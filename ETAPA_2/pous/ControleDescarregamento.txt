IF VAGAO_POSICIONADO AND (NOT NIVEL_ALTO_SILO2) THEN
    CMD_INICIAR_DESCARREGAMENTO := TRUE;
ELSE
    ALARME_ALIMENTADOR_SAIDA := TRUE;
END_IF

IF CMD_INICIAR_DESCARREGAMENTO THEN
    IF (PESO_SILO1 > PESO_SILO2) THEN
            DESVIADOR_MANGUEIRA_POS := 2;
            VENTILACAO_SILO2 := TRUE;
            ALIMENTADOR_MOEGA_ON := TRUE;
            TRANSPORTADOR_ENTRADA_ON := TRUE;
            inicio_descarregamento := TIME();

			IF (ALARME_NIVEL_ALTO_SILO2) THEN
			   	DESVIADOR_MANGUEIRA_POS :=1;
			END_IF
            // L처gica de controle da quantidade de gr찾os
            IF (PESO_SILO2 >= CAPACIDADE_SILO) THEN
                ALIMENTADOR_MOEGA_ON := FALSE;
                TRANSPORTADOR_ENTRADA_ON := FALSE;
            END_IF

            tempo := TIME() - inicio_descarregamento;
            IF (tempo >= T#2s) THEN
                TRANSPORTADOR_ENTRADA_ON := FALSE;
            END_IF
            IF (tempo >= T#5s) THEN
                VENTILACAO_SILO2 := FALSE;
            END_IF
        ELSE
            DESVIADOR_MANGUEIRA_POS := 1;
            VENTILACAO_SILO1 := TRUE;
        END_IF
    ELSE
        IF (NOT NIVEL_ALTO_SILO1) THEN
            DESVIADOR_MANGUEIRA_POS := 1;
            VENTILACAO_SILO1 := TRUE;
            ALIMENTADOR_MOEGA_ON := TRUE;
            TRANSPORTADOR_ENTRADA_ON := TRUE;
            inicio_descarregamento := TIME();

			IF (ALARME_NIVEL_ALTO_SILO1) THEN
			   	DESVIADOR_MANGUEIRA_POS :=2;
				VENTILACAO_SILO2 := TRUE;
                ALIMENTADOR_MOEGA_ON := TRUE;
                TRANSPORTADOR_ENTRADA_ON := TRUE;
                inicio_descarregamento := TIME();
			END_IF
            // L처gica de controle da quantidade de gr찾os
            IF (PESO_SILO1 >= CAPACIDADE_SILO) THEN
                ALIMENTADOR_MOEGA_ON := FALSE;
                TRANSPORTADOR_ENTRADA_ON := FALSE;
            END_IF

            tempo := TIME() - inicio_descarregamento;
            IF (tempo >= T#2s) THEN
                TRANSPORTADOR_ENTRADA_ON := FALSE;
            END_IF
            IF (tempo >= T#5s) THEN
                VENTILACAO_SILO1 := FALSE;
            END_IF
        ELSE
            DESVIADOR_MANGUEIRA_POS := 2;
            VENTILACAO_SILO2 := TRUE;
        END_IF
    END_IF
END_IF
